Core Player Detection


# Core player detection using YOLOv8
class PlayerDetector:
    def __init__(self):
        self.model = YOLO('models/yolov8n.pt')
        self.confidence_threshold = 0.5

    def detect_players(self, frame):
        # Run YOLOv8 inference
        results = self.model(frame)
        
        players = []
        for detection in results[0].boxes.data:
            if detection[5] == 0:  # Class 0 is person
                confidence = detection[4]
                if confidence > self.confidence_threshold:
                    x1, y1, x2, y2 = detection[0:4]
                    players.append({
                        'bbox': (int(x1), int(y1), int(x2), int(y2)),
                        'confidence': float(confidence)
                    })
        return players

Team Recognition


# Team recognition using color analysis
class TeamRecognition:
    def __init__(self):
        self.team_colors = {
            'team1': {'lower': None, 'upper': None},
            'team2': {'lower': None, 'upper': None}
        }

    def calibrate_team_colors(self, frame, team1_roi, team2_roi):
        """Calibrate team colors from regions of interest"""
        # Convert ROIs to HSV color space
        team1_hsv = cv2.cvtColor(frame[team1_roi], cv2.COLOR_BGR2HSV)
        team2_hsv = cv2.cvtColor(frame[team2_roi], cv2.COLOR_BGR2HSV)
        
        # Calculate color ranges with tolerance
        self.team_colors['team1'] = self._calculate_color_range(team1_hsv)
        self.team_colors['team2'] = self._calculate_color_range(team2_hsv)

    def classify_player(self, player_roi):
        """Classify player's team based on jersey color"""
        hsv = cv2.cvtColor(player_roi, cv2.COLOR_BGR2HSV)
        
        # Check color matches for each team
        team1_match = cv2.inRange(hsv, 
                                self.team_colors['team1']['lower'],
                                self.team_colors['team1']['upper'])
        team2_match = cv2.inRange(hsv, 
                                self.team_colors['team2']['lower'],
                                self.team_colors['team2']['upper'])
        
        # Return team with highest color match
        team1_score = np.sum(team1_match)
        team2_score = np.sum(team2_match)
        
        return 'team1' if team1_score > team2_score else 'team2'

Offside Detection


# Offside analysis system
class OffsideAnalyzer:
    def __init__(self):
        self.field_detector = FieldLineDetector()
        self.perspective_transform = None

    def detect_field_lines(self, frame):
        """Detect field lines using Hough transform"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        edges = cv2.Canny(gray, 50, 150)
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 100,
                               minLineLength=100, maxLineGap=10)
        return lines

    def check_offside(self, players, ball_position):
        """Check for offside positions"""
        if self.perspective_transform is None:
            return None
            
        # Transform player positions to bird's eye view
        transformed_positions = cv2.perspectiveTransform(
            players, self.perspective_transform)
        
        # Get last defender position
        defenders = [p for p in transformed_positions if p['team'] == 'defending']
        last_defender_y = max(d['position'][1] for d in defenders)
        
        # Check attacking players
        offside_players = []
        for player in transformed_positions:
            if (player['team'] == 'attacking' and 
                player['position'][1] < last_defender_y and
                player['position'][1] < ball_position[1]):
                offside_players.append(player)
                
        return offside_players

# Main GUI implementation
class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("VAROMATIC+ Analysis")
        self.setup_ui()

    def setup_ui(self):
        # Create central widget and layout
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        layout = QVBoxLayout(central_widget)

        # Add video display widget
        self.video_widget = VideoWidget()
        layout.addWidget(self.video_widget)

        # Add control panel
        self.control_panel = ControlPanel()
        layout.addWidget(self.control_panel)

        # Add analysis panel
        self.analysis_panel = AnalysisPanel()
        layout.addWidget(self.analysis_panel)

        # Connect signals
        self.control_panel.analysis_toggled.connect(self.toggle_analysis)
        self.video_widget.frame_processed.connect(self.update_analysis)

    def process_frame(self):
        """Main processing loop"""
        ret, frame = self.video_capture.read()
        if not ret:
            return
            
        # Detect players
        players = self.player_detector.detect_players(frame)
        
        # Classify teams
        for player in players:
            player['team'] = self.team_recognition.classify_player(
                player['roi'])
        
        # Check offside
        offside_players = self.offside_analyzer.check_offside(
            players, self.ball_position)
        
        # Update display
        self.draw_results(frame, players, offside_players)
        self.video_widget.update_frame(frame)
